// Control de iluminación automático (LDR + LED)
void setup() {
  pinMode(13, OUTPUT); // Configura el LED integrado
}

void loop() {
  if (analogRead(A0) < 300) { 
    digitalWrite(13, HIGH); // Enciende si hay oscuridad
  } else {
    digitalWrite(13, LOW);  // Apaga si hay luz
  }
}
